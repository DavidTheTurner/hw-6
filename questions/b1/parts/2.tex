\question{(B1.2)}{
    Let $\mathbf{GL}^+(n,\reals)$ be the subgroup
    of $\mathbf{GL}(n,\reals)$ consisting of all
    matrices $A$ such that $\det(A) > 0$.
    Consider the function
    $\mapdef{h}{\mathbf{GL}^{+}(n+1)}{\mathbf{S}(n+1)}$, given by
    \[
        h(A) = \transpos{A}JA - J,
    \]
    where $\mathbf{S}(n+1)$ denotes the space of $(n+1)\times (n+1)$ symmetric matrices.
    Prove that
    \[
        dh_A(H) =  \transpos{A}JH +  \transpos{H}JA
    \]
    for any matrix $H\in \mathrm{M}_{n+1}(\reals)$.
    \\\medskip
    Prove that $dh_A$ is surjective for all $A\in \mathbf{SO}(n, 1)$.
    \\\medskip
    {\bf Remark: }
    Parts (1) and (2) can be used to prove that $\mathbf{SO}(n, 1)$ is a smooth manifold of dimension $\frac{n(n+1)}{2}$.
}
\begin{proof}
    We'll claim that $dh_A(H) = \transpos{A}JH + \transpos{H}JA$, this gives us
    \begin{align*}
        \varepsilon(H)\norm{H} & = h(A + H) - h(A) - (\transpos{A}JH + \transpos{H}JA)
        \\ & = \transpos{(A + H)}J(A + H) - J - (\transpos{A}JA - J) - (\transpos{A}JH + \transpos{H}JA)
        \\ & = (\transpos{A} + \transpos{H})(JA + JH) - J - \transpos{A}JA + J - \transpos{A}JH - \transpos{H}JA
        \\ & = \transpos{A}JA + \transpos{H}JA + \transpos{A}JH + \transpos{H}JH - J - \transpos{A}JA + J - \transpos{A}JH - \transpos{H}JA
        \\ & = \transpos{H}JH.
    \end{align*}
    From here we can ascertain
    \[
        |\varepsilon(H)| \leq \norm{\frac{\transpos{H}JH}{\norm{H}}} \leq \frac{\norm{\transpos{H}}\norm{J}\norm{H}}{\norm{H}} = \norm{\transpos{H}}\norm{J} = \norm{H}\norm{J}
    \]
    and so
    \[
        \lim_{H \to \infty}\varepsilon(H) = 0
    \]
    and we can conclude that
    \[
        dh_A(H) = \transpos{A}JH + \transpos{H}JA
    \]
\end{proof}
