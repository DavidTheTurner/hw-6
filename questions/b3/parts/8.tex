\question{(B3.8)}{
    Let $\mathbf{GL}^+(n,\reals)$ be the subgroup
    of $\mathbf{GL}(n,\reals)$ consisting of all
    matrices $A$ such that $\det(A) > 0$.
    It can be shown that this subgroup is open in $\mathrm{M}_n(\reals)$.
    Consider the function
    $\mapdef{\ell}{\mathbf{GL}^+(n,\reals)}{\reals}$ given by
    \[
        \ell(A) = \log\det(A).
    \]
    Prove that
    \[
        d\ell_A(B) = \mathrm{tr}(A^{-1}B)
    \]
    for all $A\in \mathbf{GL}^+(n,\reals)$ and all
    $B\in \mathrm{M}_n(\reals)$.\\
    \remark
    The function $\log\det$ is a {\it barrier\/} function used in
    convex optimization.
}
\begin{proof}
    We begin with the chain rule, and using the result of problem B3.7, we get
    \begin{align*}
        d\,\ell_A & = d\,log(det(A))\circ d\,det(A)
        \\ & = \frac{1}{det(A)}tr(\widetilde{A}H)
    \end{align*}
    Plugging in $B$, we see
    \begin{align*}
        d\,\ell_A(B) & = \frac{1}{det(A)}tr(\widetilde{A}B)
        \\ & = tr(\frac{1}{det(A)}\widetilde{A}B)
        \\ & = tr(A^{-1}\widetilde{A}^{-1}\widetilde{A}B)
        \\ & = tr(A^{-1}B)
    \end{align*}
    as desired.
\end{proof}
