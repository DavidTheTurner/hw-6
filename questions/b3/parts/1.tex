\question{(B3.1)}{
    Use the chain rule to prove that
    \[
        d\,\mathrm{det}_I(B) = (\det\circ\, \gamma)'(0),
    \]
    where $d\,\mathrm{det}_I$ is the derivative of the determinant function
    $\mapdef{\mathrm{det}}{\mathrm{M}_n(\reals)}{\reals}$ at $I$
    (the identity matrix).
}
\begin{proof}
    Using the chain rule, we get
    \begin{align*}
        (\det\circ\, \gamma)'(0) & = d\,\det(\gamma(0))\circ\ \gamma'(0)
        \\ & = d\,\det(I)\circ\ B
        \\ & = d\,\mathrm{det}_I(B)
    \end{align*}
    as desired.
\end{proof}
