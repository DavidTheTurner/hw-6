\question{(B3.6)}{
    Prove that for {\it any\/} matrix $A\in \mathrm{M}_n(\reals)$,
    not necessarily  invertible, there is a convergent sequence
    $(A_k)_{k\geq 1}$ of {\it invertible\/} matrices $A_k\in
    \mathbf{GL}(n, \reals)$ whose limit is $A$.
    To prove this, it is convenient to use the
    Frobenius norm or the operator $2$-norm (the spectral norm).
    You need to construct a sequence of  invertible matrices $A_k$ such that
    \[
        \lim_{k \mapsto \infty} \norme{A - A_k} = 0.
    \]\\
    \hint
    Use a convenient factorization of $A$.
}
\begin{proof}
    We can use the SVD to construct our sequence. Let the SVD of $A$ be $VD\transpos{U}$
    then define our sequence of invertible matrices $(A_k)_{k\geq 1}$ as $A_k = V(D + \frac{1}{k}I)\transpos{U}$.
    Using the Frobenius norm, we get
    \begin{align*}
        \lim_{k \mapsto \infty} \norme{A - A_k} & = \lim_{k \mapsto \infty} \norme{VD\transpos{U} - V(D + \frac{1}{k}I)\transpos{U}}
        \\ & = \lim_{k \mapsto \infty} \norme{VD\transpos{U} - V(D + \frac{1}{k}I)\transpos{U}}
        \\ & = \lim_{k \mapsto \infty} \norme{V(D - (D + \frac{1}{k}I))\transpos{U}}
        \\ & = \lim_{k \mapsto \infty} \norme{D - (D + \frac{1}{k}I)} \tag{Frobenius norm is invariant to orthogonal matrices}
        \\ & = 0
    \end{align*}
    as desired.
\end{proof}
